drop table if exists NinetiesMusic;

create table NinetiesMusic (
	track text primary key,
	artistName string,
	yearReleased int CHECK (1989 > yearReleased < 2000),
	timesPlayed int,
	award string CHECK (award = 'Platinum') NOT NULL
);
